function getLCString(aStr, bStr) {
    var len = aStr.length < bStr.length ? aStr.length : bStr.length,
        max = -1;
 
    for (var i = 0; i < len; i++) {
        if (aStr.charAt(i) == bStr.charAt(i) && i - max == 1) {
